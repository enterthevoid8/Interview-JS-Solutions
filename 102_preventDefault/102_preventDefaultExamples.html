preventDefault() Examples

1. Preventing Form Submission

Please fill out all fields.

      
document.querySelector('#example-form').addEventListener('submit', function (event) {
  event.preventDefault(); // Prevent the form from submitting

  const username = event.target.username.value.trim();
  const password = event.target.password.value.trim();

  if (!username || !password) {
    document.getElementById('form-error').style.display = 'block';
  } else {
    console.log('Form Data:', { username, password });
    alert('Form submitted successfully!');
  }
});

document.querySelector('#example-link').addEventListener('click', function (event) {

3. Preventing Default Keyboard Behavior

      
document.querySelector('#example-input').addEventListener('keydown', function (event) {
  if (event.key === 'Enter') {
    event.preventDefault(); // Prevent form submission on Enter key
    console.log('Enter key pressed, but default behavior prevented.');
    alert('Enter key pressed, but default behavior prevented.');
  }
});

5. Preventing Drag-and-Drop Behavior

document.addEventListener('dragover', function (event) {
  event.preventDefault(); // Prevent the default drag-and-drop behavior
});

document.addEventListener('drop', function (event) {
  event.preventDefault(); // Prevent the default drop behavior
  console.log('File dropped:', event.dataTransfer.files);
  alert('File dropped! Check the console for details.');
});

6. Difference Between preventDefault() and stopPropagation()

Click Me 
      
document.querySelector('#example-link-2').addEventListener('click', function (event) {
  event.preventDefault(); // Prevent navigation
  event.stopPropagation(); // Stop the event from bubbling up
  console.log('Link clicked, but default behavior and propagation stopped.');
  alert('Link clicked, but default behavior and propagation stopped.');
});
